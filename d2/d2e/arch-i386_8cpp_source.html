60 const char* cpuGetModel()
61 {
62  // The CPU model is broken up across 3 different calls, each using
63  // EAX, EBX, ECX, and EDX to store the string, so we basically
64  // are appending all 4 register values to this char array each time.
65  static char model[48];
66  i386::cpuid(0x80000002, (int*)(model));
67  i386::cpuid(0x80000003, (int*)(model + 16));
68  i386::cpuid(0x80000004, (int*)(model + 32));
69  return model;
70 }
