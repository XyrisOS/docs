33 void timer_init(uint32_t freq) {
34  /* Install the function we just wrote */
35  register_interrupt_handler(IRQ0, timer_callback);
36  /* Get the PIT value: hardware clock at 1193180 Hz */
37  uint32_t divisor = 1193180 / freq;
38  uint8_t low = (uint8_t)(divisor & 0xFF);
39  uint8_t high = (uint8_t)((divisor >> 8) & 0xFF);
40  /* Send the command */
41  writeByte(TIMER_COMMAND_PORT, 0x36);
