76 /**
77  * @brief This is the Xyris kernel entry point. This function is called directly
78  * from the assembly written in boot.S located in Arch/i686/boot.S. The pragma
79  * allows this function to be declared without needing a former declaration.
80  */
81 extern "C" void kernelEntry(void* info, uint32_t magic)
82 {
83  // Initialize the CPU
84  Arch::CPU::init();
85  // Initialize devices
86  Arch::CPU::criticalRegion(devInit);
87  // Initialize info from bootloader
88  Boot::Handoff handoff(info, magic);
89  Memory::init(handoff.MemoryMap());
90  Graphics::init(handoff.FramebufferInfo());
91  // Print the splash screen to show we've booted into the kernel properly.
92  printSplash();
93  // Print some info to show we did things right
94  Time::TimeDescriptor time;
95  Console::printf(DBG_INFO "UTC: %i/%i/%i %i:%i\n",
96  time.getMonth(),
97  time.getDay(),
98  time.getYear(),
99  time.getHour(),
100  time.getMinutes());
101  // Get the CPU vendor and model data to print
102  const char* vendor = Arch::CPU::vendor();
103  const char* model = Arch::CPU::model();
104  Console::printf(DBG_INFO "%s %s\n", vendor, model);
105  RS232::printf("%s\n%s\n", vendor, model);
106 
107  tasks_init();
108  struct task compute, status, spinner;
109  tasks_new(Apps::find_primes, &compute, TASK_READY, "prime_compute");
110  tasks_new(Apps::show_primes, &status, TASK_READY, "prime_display");
111  tasks_new(Apps::spinner, &spinner, TASK_READY, "spinner");
112  // Now that we're done make a joyful noise
113  bootTone();
114 
115  // Keep the kernel task alive.
116  tasks_block_current(TASK_PAUSED);
117  panic("Kernel terminated unexpectedly!");
118 }
