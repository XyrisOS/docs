91 static void callback(struct registers *regs) {
92  (void)regs;
93  // Grab the input character
94  char in = read_byte();
95  // Change carriage returns to newlines
96  if (in == '\r') {
97  in = '\n';
98  }
99  // Create a string and print it so that the
100  // user can see what they're typing.

102  printf("%s", str);
103  // Add the character to the circular buffer
104  ring.Enqueue(in);

140 size_t read(char* buf, size_t count) {
141  size_t bytes = 0;
142  mutex_rs232.Lock();
143  for (size_t idx = 0; idx < count && !ring.IsEmpty(); idx++)
144  {
145  buf[idx] = ring.Dequeue();
146  bytes++;
147  }
148  mutex_rs232.Unlock();
149  return bytes;
150 }

152 size_t write(const char* buf, size_t count) {
153  // Wait for previous transfer to complete
154  while (is_transmit_empty() == 0);
155  size_t bytes = 0;
156  for (size_t idx = 0; idx < count; idx++) {
157  writeByte(rs_232_port_base + RS_232_DATA_REG, buf[idx]);
158  bytes++;
159  }
160  return bytes;
