24 void gdt_set_gate(uint8_t num, uint64_t base, uint64_t limit, uint16_t flags)
25 {
26  // 32-bit address space
27  // Now we have to squeeze the (32-bit) limit into 2.5 regiters (20-bit).
28  // This is done by discarding the 12 least significant bits, but this
29  // is only legal, if they are all ==1, so they are implicitly still there
30 
31  // so if the last bits aren't all 1, we have to set them to 1, but this
32  // would increase the limit (cannot do that, because we might go beyond
33  // the physical limit or get overlap with other segments) so we have to
34  // compensate this by decreasing a higher bit (and might have up to
35  // 4095 wasted bytes behind the used memory)
36  uint64_t descriptor = 0;
37 
38  // Create the high 32 bit segment
39  descriptor = base & 0xFF000000; // base direct map
40  descriptor |= (base >> 16) & 0x000000FF; // base 23-16 : 7-0
41  descriptor |= (flags << 8) & 0x00F0FF00; // flags 16-11 : 24-19 7-0 : 15-8
42  descriptor |= limit & 0x000F0000; // limit direct map
43  // Shift by 32 to move to the lower half of the section
44  descriptor <<= 32;
45  // Create the low 32 bit segment
46  descriptor |= (base << 16) & 0xFFFF0000; // base 15-0 : 31-16
47  descriptor |= limit & 0x0000FFFF; // limit direct map
48  // Copy the descriptor value into our GDT entries array
49  memcpy(&gdt_entries[num], &descriptor, sizeof(uint64_t));
50 }
