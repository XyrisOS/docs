107 static int putchar(unsigned c, void** ptr)
108 {
109  (void)ptr;
110  // Check the ANSI state
111  switch (ansiState) {
112  case Normal: // print the character out normally unless it's an ESC
113  if (c != ESC)
114  break;
115  ansiState = Esc;
116  goto end;
117  case Esc: // we got an ESC, now we need a left square bracket
118  if (c != '[')
119  break;
120  ansiState = Bracket;
121  goto end;
122  case Bracket: // we're looking for a value/command char now
123  if (c >= '0' && c <= '9') {
124  ansiVal = (uint16_t)(c - '0');
125  ansiState = Value;
126  goto end;
127  } else if (c == 's') { // Save cursor position attribute
128  ansiCursorX = cursorX;
129  ansiCursorY = cursorY;
130  goto normal;
131  } else if (c == 'u') { // Restore cursor position attribute
132  cursorX = ansiCursorX;
133  cursorY = ansiCursorY;
134  goto normal;
135  }
136  break;
137  case Value:
138  if (c == ';') { // the semicolon is a value separator
139  // enqueue the value here
140  PUSH_VAL(ansiVal);
141  ansiState = Bracket;
142  ansiVal = 0;
143  } else if (c == 'm') { // Set color/text attributes command
144  PUSH_VAL(ansiVal);
145  // take action here
146  // iterate through all values
147  while (ansiValuesIdx > 0) {
148  ansiVal = (uint16_t)POP_VAL();
149  if (ansiVal == 0) {
150  // Reset code will just reset to whatever was specified in tty_clear().
151  colorFore = resetFore;
152  colorBack = resetBack;
153  } else if (ansiVal >= Black && ansiVal <= White) {
154  colorFore = (vgaColor)ansiVGATable[ansiVal - Black];
155  } else if (ansiVal >= (Black + Background) && ansiVal <= (White + Background)) {
156  colorBack = (vgaColor)ansiVGATable[ansiVal - (Black + Background)];
157  } else if (ansiVal >= BrightBlack && ansiVal <= BrightWhite) {
158  colorFore = (vgaColor)ansiVGATable[ansiVal - BrightBlack + 8];
159  } else if (ansiVal >= (BrightBlack + Background) && ansiVal <= (BrightWhite + Background)) {
160  colorBack = (vgaColor)ansiVGATable[ansiVal - (BrightBlack + Background) + 8];
161  } // else it was an unknown code
162  }
163  goto normal;
164  } else if (c == 'H' || c == 'f') { // Set cursor position attribute
165  PUSH_VAL(ansiVal);
166  // the proper order is 'line (y);column (x)'
167  if (ansiValuesIdx > 2) {
168  goto error;
169  }
170  cursorX = (uint8_t)POP_VAL();
171  cursorY = (uint8_t)POP_VAL();
172  goto normal;
173  } else if (c == 'J') { // Clear screen attribute
174  // The proper code is ESC[2J
175  if (ansiVal != 2) {
176  goto error;
177  }
178  // Clear by resetting the double buffer and swapping.
179  // TODO: Find a better way to do this?
180  debugf("Clearing screen\n");
181  Graphics::resetDoubleBuffer();
182  Graphics::swap();
183  } else if (c >= '0' && c <= '9') { // just another digit of a value
184  ansiVal = (uint16_t)(ansiVal * 10 + (uint16_t)(c - '0'));
185  } else
186  break; // invald code, so just return to normal
187  // we hit one of the cases so return
188  goto end;
189  }
190  // we fell through some way or another so just reset to Normal no matter what
191  ansiState = Normal;
192  switch (c) {
193  // Backspace
194  case 0x08:
195  if (cursorX > 0) {
196  cursorX--;
197  }
198  break;
199  // Newline
200  case '\n':
201  cursorX = 0;
202  cursorY++;
203  break;
204  case '\t':
205  while (++cursorX % TAB_WIDTH)
206  ;
207  break;
208  // Carriage return
209  case '\r':
210  cursorX = 0;
211  break;
212  // Anything else (visible characters)
213  default:
214  Graphics::Font::Draw(c, cursorX++, cursorY, colorFore, colorBack);
215  }
216 
217  // Move to the next line
218  // TODO: Get width of "screen" (replace 80 with width) (#275)
219  if (cursorX >= 80) {
220  debugf("Move to the next line\n");
221  cursorX = 0;
222  cursorY++;
223  }
224  // Clear the screen
225  // TODO: Get height of "screen" (replace 25 with height) (#275)
226  if (cursorY >= 25) {
227  debugf("Shift up the screen\n");
228  //TODO: Shift text up and reset the bottom line
229  cursorX = 0;
230  cursorY = 25 - 1;
231  }
232  goto end;
233 error:
234  // Reset stack index
235  CLEAR_VALS();
236  // Return to normal
237  ansiState = Normal;
238  ansiVal = 0;
239  return -1;
240 normal:
241  ansiState = Normal;
242  ansiVal = 0;
243 end:
244  return 0;
245 }
