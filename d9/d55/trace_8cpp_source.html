19 void stack_trace(size_t max)
20 {
21  // Define our stack
22  struct stackframe* stk;
23  asm volatile("movl %%ebp, %0"
24  : "=r"(stk)::);
25  Console::printf("\033[0;%iH \033[31mStack trace:\033[0m\n", (X86_TTY_WIDTH - 16));
26  RS232::printf("%s", "\n\033[31mStack trace:\033[0m\n");
27  char buf[32];
28  for (size_t frame = 0; stk != NULL && frame < max; ++frame) {
29  // Unwind to previous stack frame
30  ksprintf(buf, "0x%08X\n", stk->eip);
31  Console::printf("\033[%i;%iH %s", (frame + 1), (X86_TTY_WIDTH - 16), buf);
32  RS232::printf("%s", buf);
33  // Check whether the address is in memory or not
34  if (!page_is_present((size_t)stk->ebp))
35  break;
36  stk = stk->ebp;
37  }
38 }
