126  /**
127  * @brief Finds a range of `count` clear bits and returns the starting position.
128  *
129  * @param count Number of clear bits desired
130  * @param isSet If true, find the first range of `count` bits that are set.
131  * @return size_t Position of the first bit with desired range and polarity.
132  * If all bits are polarized, SIZE_MAX is returned.
133  */
134  ALWAYS_INLINE size_t FindFirstRange(size_t count, bool isSet)
135  {
136  size_t checkLow, checkHigh, check, idx, offset;
137  size_t mask = ((size_t)1 << count) - (size_t)1;
138  for (size_t i = 0UL; i < S - count; i++) {
139  idx = Index(i);
140  offset = Offset(i);
141  checkLow = map[idx] >> offset;
142  checkHigh = offset ? map[idx + 1] << (TypeSize() - offset) : 0;
143  check = checkLow | checkHigh;
144 
145  if ((check & mask) == isSet)
146  return i;
147  }
148  return SIZE_MAX;
149  }
