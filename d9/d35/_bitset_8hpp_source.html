112  /**

114  *
115  * @param count Number of clear bits desired
116  * @param isSet If true, find the first range of `count` bits that are set.
117  * @return size_t Position of the first bit with desired range and polarity.
118  * If all bits are polarized, SIZE_MAX is returned.
119  */
120  ALWAYS_INLINE size_t FindFirstRange(size_t count, bool isSet)
121  {
122  size_t checkLow, checkHigh, check, idx, offset;
123  size_t mask = ((size_t)1 << count) - (size_t)1;
124  for (size_t i = 0UL; i < S - count; i++) {
125  idx = Index(i);
126  offset = Offset(i);
127  checkLow = map[idx] >> offset;
128  checkHigh = offset ? map[idx + 1] << (TypeSize() - offset) : 0;
129  check = checkLow | checkHigh;
130 
131  if ((check & mask) == isSet)
132  return i;
133  }
134  return SIZE_MAX;
135  }
