47 extern "C" void
48 SECTION(".early_text")
49 OPTIMIZE(0)
50 early_panic(const char *str) {
51  volatile uint16_t* where;
52  int x = 0;
53  int y = 0;
54  // Clear the screen
55  for (int i = 0; i < TTY_WIDTH; i++) {
56  for (int j = 0; j < TTY_HEIGHT; j ++) {
57  where = biosVGABuffer + (j * TTY_WIDTH + i);
58  *where = VGA_CHAR(' ', VGA_COLOR(BIOS_Black, BIOS_White));
59  }
60  }
61  // For each character in the string
62  for (int i = 0; str[i] != '\0'; ++i) {
63  switch(str[i]) {
64  // Newline
65  case '\n':
66  x = 0;
67  y++;
68  break;
69  // Anything else
70  default:
71  where = biosVGABuffer + (y * TTY_WIDTH + x);
72  *where = VGA_CHAR(str[i], VGA_COLOR(BIOS_Red, BIOS_White));
73  x++;
74  break;
75  }
76  }
77 }
