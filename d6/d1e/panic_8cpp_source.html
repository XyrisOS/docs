71 NORET void panic(struct registers* regs, const char* file, uint32_t line, const char* func)
72 {
73  asm volatile("cli");
74  // Print the panic cow and exception description
75  printPanicScreen();
76  char msg[128];
77  ksprintf(
78  msg,
79  "Exception: %i (%s)\n\n",
80  regs->int_num,
81  exception_descriptions[regs->int_num]);
82  // Print to VGA and serial
83  Console::printf("%s", msg);
84  RS232::printf("%s", msg);
85  // Check if we have an error code and print
86  if (regs->err_code) {
87  ksprintf(
88  msg,
89  "Error code: %i",
90  regs->err_code);
91  // Print to VGA and serial
92  Console::printf("%s", msg);
93  RS232::printf("%s", msg);
94  }
95  // Print out register values
96  panicPrintRegister(regs);
97  // A page fault has occurred.
98  // The faulting address is stored in the CR2 register.
99  size_t faulting_address;
100  asm volatile("mov %%cr2, %0"
101  : "=r"(faulting_address));
102  // The error code gives us details of what happened.
103  int present = !(regs->err_code & 0x1); // Page not present
104  int rw = regs->err_code & 0x2; // Write operation?
105  int us = regs->err_code & 0x4; // Processor was in user-mode?
106  int reserved = regs->err_code & 0x8; // Overwritten CPU-reserved bits of page entry?
107  int id = regs->err_code & 0x10; // Caused by an instruction fetch?
108  // If we have a page fault, print out page fault info
109  if (regs->int_num == ISR_PAGE_FAULT) {
110  // Output an error message.
111  const char* real = (present ? "present " : "missing ");
112  const char* rws = (rw ? "reading " : "writing ");
113  const char* uss = (us ? "user-mode " : "kernel ");
114  const char* avail = (reserved ? "reserved" : "available");
115  // Now that we assigned all of our string, put together the message
116  ksprintf(
117  msg,
118  "Page fault (%s%s%s%s) at 0x%08X (id -> %i)\n",
119  real,
120  rws,
121  uss,
122  avail,
123  faulting_address,
124  id);
125  // Print to VGA and serial
126  Console::printf("%s", msg);
127  RS232::printf("%s", msg);
128  }
129  panicPrintSource(file, line, func);
130  stack_trace(16);
131  // Halt the CPU
132  while (true) {
133  asm("hlt");
134  }
135 }
