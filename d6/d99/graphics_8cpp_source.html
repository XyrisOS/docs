37 void init()
38 {
39  // Get the framebuffer info
40  if (!(info = getFramebuffer()))
41  return;
42  // Ensure valid info is provided
43  if (!info->getAddress())
44  return;
45  // Map in the framebuffer
46  RS232::printf("Mapping framebuffer...\n");
47  map_kernel_range_virtual((uintptr_t)info->getAddress(), (uintptr_t)info->getAddress() + (info->getPitch() * info->getHeight()));
48  // Alloc the backbuffer
49  backbuffer = malloc(info->getPitch() * info->getHeight());
50  memcpy(backbuffer, info->getAddress(), info->getPitch() * info->getHeight());
51 
52  initialized = true;
53 }

55 void pixel(uint32_t x, uint32_t y, uint32_t color)
56 {
57  // Ensure framebuffer information exists
58  if (!initialized)
59  return;
60  if ((x <= info->getWidth()) && (y <= info->getHeight())) {
61  // Special thanks to the SkiftOS contributors.
62  uint8_t* pixel = (uint8_t*)backbuffer + (y * info->getPitch()) + (x * info->getPixelWidth());
63  // Pixel information
64  pixel[0] = (color >> info->getBlueMaskShift()) & 0xff; // B
65  pixel[1] = (color >> info->getGreenMaskShift()) & 0xff; // G
66  pixel[2] = (color >> info->getRedMaskShift()) & 0xff; // R
67  // Additional pixel information
68  if (info->getPixelWidth() == 4)
69  pixel[3] = 0x00;
70  }
71 }
