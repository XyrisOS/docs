63 void paging_init(uint32_t page_count) {
64  machine_page_count = page_count;
65  // we can set breakpoints or make a futile attempt to recover.
66  register_interrupt_handler(ISR_PAGE_FAULT, mem_page_fault);
67  // init our structures
68  paging_init_dir();

83 static inline void map_kernel_page_table(uint32_t pd_idx, struct page_table *table) {
84  page_dir_virt[pd_idx] = table;
85  page_dir_phys[pd_idx] = {
86  .present = 1,
87  .read_write = 1,
88  .usermode = 0,
89  .write_through = 0,
90  .cache_disable = 0,
91  .accessed = 0,
92  .ignored_a = 0,
93  .page_size = 0,
94  .ignored_b = 0,
95  // compute the physical address of this page table
96  // the virtual address is obtained with the & operator and
97  // the offset is applied from the load address of the kernel
98  // we must shift it over 12 bits because we only care about
99  // the highest 20 bits for the page table
100  .table_addr = KADDR_TO_PHYS((uint32_t)table) >> 12
101  };
102 }

104 static void paging_init_dir() {
105  // For every page in kernel memory
106  for (int i = 0; i < PAGE_ENTRIES - 1; i++) {
107  map_kernel_page_table(i, &page_tables[i]);
108  // clear out the page tables
109  for (int j = 0; j < PAGE_ENTRIES; j++) {
110  page_tables[i].pages[j] = (struct page_table_entry){ /* ZERO */ };
111  }
112  }
113  // recursively map the last page table to the page directory
114  map_kernel_page_table(PAGE_ENTRIES - 1, (struct page_table*)&page_dir_phys[0]);
115  for (uint32_t i = PAGE_ENTRIES * (PAGE_ENTRIES - 1); i < PAGE_ENTRIES * PAGE_ENTRIES; i++) {
116  mapped_pages.Set(i);
117  }
118  // store the physical address of the page directory for quick access
119  page_dir_addr = KADDR_TO_PHYS((uint32_t)&page_dir_phys[0]);
120 }

122 void map_kernel_page(virtual_address_t vaddr, uint32_t paddr) {
123  // Set the page directory entry (pde) and page table entry (pte)
124  uint32_t pde = vaddr.page_dir_index;
125  uint32_t pte = vaddr.page_table_index;
126  // If the page's virtual address is not aligned
127  if (vaddr.page_offset != 0) {
128  PANIC("Attempted to map a non-page-aligned virtual address.\n");
129  }
130  page_table_entry *entry = &(page_tables[pde].pages[pte]);
131  // Print a debug message to serial
132  if (is_mapping_output_enabled) {
133  debugf("map 0x%08x to 0x%08x, pde = 0x%08x, pte = 0x%08x\n", paddr, vaddr.val, pde, pte);
134  }
135  // If the page is already mapped into memory
136  if (entry->present) {
137  if (entry->frame == paddr >> 12) {
138  // this page was already mapped the same way
139  return;
140  }
141 
142  PANIC("Attempted to map already mapped page.\n");
143  }
144  // Set the page information
145  page_tables[pde].pages[pte] = {
146  .present = 1, // The page is present
147  .read_write = 1, // The page has r/w permissions
148  .usermode = 0, // These are kernel pages
149  .write_through = 0, // Disable write through
150  .cache_disable = 0, // The page is cached
151  .accessed = 0, // The page is unaccessed
152  .dirty = 0, // The page is clean
153  .page_att_table = 0, // The page has no attribute table
154  .global = 0, // The page is local
155  .unused = 0, // Ignored
156  .frame = paddr >> 12 // The last 20 bits are the frame
157  };
158  // Set the associated bit in the bitmaps
159  mapped_mem.Set(paddr >> 12);
160  mapped_pages.Set(vaddr.val >> 12);
161 }

185 static inline void paging_enable() {
186  size_t cr0;
187  asm volatile("mov %%cr0, %0": "=b"(cr0));
188  // 0x80000000 = 0b10000000000000000000000000000000
189  // The most significant bit signifies whether to
190  // enable or disable paging within control register 0.
191  cr0 |= 0x80000000;
192  asm volatile("mov %0, %%cr0":: "b"(cr0));
193 }

195 static inline void paging_disable() {
196  size_t cr0;
197  asm volatile("mov %%cr0, %0": "=b"(cr0));
198  // 0x80000000 = 0b10000000000000000000000000000000
199  // The most significant bit signifies whether to
200  // enable or disable paging within control register 0.
201  // In this case we set the opposite (~) so the result
202  // is 0b01111111111111111111111111111111
203  cr0 &= ~(0x80000000U);
204  asm volatile("mov %0, %%cr0":: "b"(cr0));
205 }

220  * map in a new page. if you request less than one page, you will get exactly one page

222 void* get_new_page(uint32_t size) {
223  mutex_paging.Lock();
224  uint32_t page_count = (size / PAGE_SIZE) + 1;
225  uint32_t free_idx = find_next_free_virt_addr(page_count);
226  if (free_idx == SIZE_MAX) return NULL;
227  for (uint32_t i = free_idx; i < free_idx + page_count; i++) {
228  uint32_t phys_page_idx = find_next_free_phys_page();
229  if (phys_page_idx == SIZE_MAX) return NULL;
230  map_kernel_page(VADDR((uint32_t)i * PAGE_SIZE), phys_page_idx * PAGE_SIZE);
231  }
232  mutex_paging.Unlock();
233  return (void *)(free_idx * PAGE_SIZE);
234 }

237  mutex_paging.Lock();
238  uint32_t page_count = (size / PAGE_SIZE) + 1;
239  uint32_t page_index = (uint32_t)page >> 12;
240  for (uint32_t i = page_index; i < page_index + page_count; i++) {
241  // TODO: need locking here (maybe make a paging lock)
242  mapped_pages.Clear(i);
243  // how much more UN-readable can we make this?? (pls, i need to know...)
244  //*(uint32_t*)((uint32_t)page_tables + i * 4) = 0;
245  // this is the same as the line above
246  struct page_table_entry *pte = &(page_tables[i / PAGE_ENTRIES].pages[i % PAGE_ENTRIES]);
247  // the frame field is actually the page frame's index
248  // basically it's frame 0, 1...(2^21-1)
249  mapped_mem.Clear(pte->frame);
250  // zero it out to unmap it
251  *pte = { /* Zero */ };
252  // clear that tlb
253  invalidate_page(page);
254  }
