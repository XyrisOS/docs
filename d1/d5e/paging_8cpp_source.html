56 void init(MemoryMap* map)
57 {
58  // we can set breakpoints or make a futile attempt to recover.
59  register_interrupt_handler(ISR_PAGE_FAULT, pageFaultCallback);
60  // populate the physical memory map based on bootloader information
61  initPhysical(map);
62  // init our structures
63  initDirectory();
64  // identity map the first 1 MiB of RAM
65  mapEarlyMem();
66  // map in our higher-half kernel
67  mapKernel();
68  // use our new set of page tables
69  Arch::Memory::setPageDirectory(Arch::Memory::pageAlign(pageDirectoryAddress));
70  // flush the tlb and we're off to the races!
71  Arch::Memory::pagingEnable();
72 }

79 static void initPhysical(MemoryMap* map)
80 {
81  size_t freeBytes = 0;
82  size_t reservedBytes = 0;
83 
84  for (size_t i = 0; i < map->Count(); i++) {
85  auto section = map->Get(i);
86  if (section.initialized()) {
87  switch (section.type())
88  {
89  case Available:
90  physical.setFree(section);
91  freeBytes += section.size();
92  break;
93  default:
94  physical.setUsed(section);
95  reservedBytes += section.size();
96  break;
97  }
98  }
99  }
100 
101  debugf("Available memory: %zu MB\n", B_TO_MB(freeBytes));
102  debugf("Reserved memory: %zu MB\n", B_TO_MB(reservedBytes));
103  debugf("Total memory: %zu MB\n", B_TO_MB(freeBytes + reservedBytes));
104 }

106 static inline void mapKernelPageTable(size_t idx, struct Arch::Memory::Table* table)
107 {
108  pageDirectoryVirtual[idx] = table;
109  pageDirectoryPhysical[idx] = {
110  .present = 1,
111  .readWrite = 1,
112  .usermode = 0,
113  .writeThrough = 0,
114  .cacheDisable = 0,
115  .accessed = 0,
116  .ignoredA = 0,
117  .size = 0,
118  .ignoredB = 0,
119  // compute the physical address of this page table the virtual address is obtained with the & operator and
120  // the offset is applied from the load address of the kernel we must shift it over 12 bits because we only
121  // care about the highest 20 bits for the page table
122  // TODO: Get rid of this shift by using ``union Address``
123  .tableAddr = KADDR_TO_PHYS((uintptr_t)table) >> 12
124  };
125 }

127 static void initDirectory()
128 {
129  // For every page in kernel memory
130  for (size_t i = 0; i < ARCH_PAGE_DIR_ENTRIES - 1; i++) {
131  mapKernelPageTable(i, &pageTables[i]);
132  // clear out the page tables
133  for (size_t j = 0; j < ARCH_PAGE_DIR_ENTRIES; j++) {
134  memset(&pageTables[i].pages[j], 0, sizeof(struct Arch::Memory::TableEntry));
135  }
136  }
137  // recursively map the last page table to the page directory
138  mapKernelPageTable(ARCH_PAGE_TABLE_ENTRIES - 1, (struct Arch::Memory::Table*)&pageDirectoryPhysical[0]);
139  for (size_t i = ARCH_PAGE_TABLE_ENTRIES * (ARCH_PAGE_TABLE_ENTRIES - 1); i < ARCH_PAGE_TABLE_ENTRIES * ARCH_PAGE_TABLE_ENTRIES; i++) {
140  mappedPages.Set(i);
141  }
142  // store the physical address of the page directory for quick access
143  pageDirectoryAddress = KADDR_TO_PHYS((uintptr_t)&pageDirectoryPhysical[0]);
144 }

146 void mapKernelPage(Arch::Memory::Address vaddr, Arch::Memory::Address paddr)
147 {
148  // Set the page directory entry (pde) and page table entry (pte)
149  size_t pde = vaddr.page().dirIndex;
150  size_t pte = vaddr.page().tableIndex;
151  // If the page's virtual address is not aligned
152  if (vaddr.page().offset != 0) {
153  panic("Attempted to map a non-page-aligned virtual address.\n");
154  }
155  Arch::Memory::TableEntry* entry = &(pageTables[pde].pages[pte]);
156  // Print a debug message to serial
157  if (is_mapping_output_enabled) {
158  debugf("map 0x%08lx to 0x%08lx, pde = 0x%08lx, pte = 0x%08lx\n", paddr.val(), vaddr.val(), pde, pte);
159  }
160  // If the page is already mapped into memory
161  if (entry->present) {
162  if (entry->frame == paddr.frame().index) {
163  // this page was already mapped the same way
164  return;
165  }
166 
167  panic("Attempted to map already mapped page.\n");
168  }
169  // Set the page information
170  pageTables[pde].pages[pte] = {
171  .present = 1, // The page is present
172  .readWrite = 1, // The page has r/w permissions
173  .usermode = 0, // These are kernel pages
174  .writeThrough = 0, // Disable write through
175  .cacheDisable = 0, // The page is cached
176  .accessed = 0, // The page is unaccessed
177  .dirty = 0, // The page is clean
178  .pageAttrTable = 0, // The page has no attribute table
179  .global = 0, // The page is local
180  .unused = 0, // Ignored
181  .frame = paddr.frame().index, // The last 20 bits are the frame
182  };
183  // Set the associated bit in the bitmaps
184  physical.setUsed(paddr);
185  mappedPages.Set(vaddr.frame().index);
186 }

224 void* newPage(size_t size)
225 {
226  pagingLock.Lock();
227  size_t page_count = PAGE_COUNT(size);
228  size_t free_idx = findNextFreeVirtualAddress(page_count);
229  if (free_idx == SIZE_MAX)
230  return NULL;
231  for (size_t i = free_idx; i < free_idx + page_count; i++) {
232  size_t phys_page_idx = physical.findNextFreePhysicalAddress();
233  if (phys_page_idx == SIZE_MAX)
234  return NULL;
235  Arch::Memory::Address phys(phys_page_idx * ARCH_PAGE_SIZE);
236  Arch::Memory::Address vaddr(i * ARCH_PAGE_SIZE);
237  mapKernelPage(vaddr, phys);
238  }
239  pagingLock.Unlock();
240  return (void*)(free_idx * ARCH_PAGE_SIZE);
241 }

243 void freePage(void* page, size_t size)
244 {
245  pagingLock.Lock();
246  size_t page_count = PAGE_COUNT(size);
247  Arch::Memory::Address addr((uintptr_t)page);
248  for (size_t i = addr.page().tableIndex; i < addr.page().tableIndex + page_count; i++) {
249  mappedPages.Reset(i);
250  // this is the same as the line above
251  struct Arch::Memory::TableEntry* pte = &(pageTables[i / ARCH_PAGE_TABLE_ENTRIES].pages[i % ARCH_PAGE_TABLE_ENTRIES]);
252  // the frame field is actually the page frame's index basically it's frame 0, 1...(2^21-1)
253  physical.setFree(pte->frame);
254  // zero it out to unmap it
255  memset(pte, 0, sizeof(struct Arch::Memory::TableEntry));
256  // clear that tlb
257  Arch::Memory::pageInvalidate(page);
258  }
259  pagingLock.Unlock();
260 }
