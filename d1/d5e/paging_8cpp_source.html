56 void init(MemoryMap* map)
57 {
58  // we can set breakpoints or make a futile attempt to recover.
59  Interrupts::registerHandler(Interrupts::EXCEPTION_PAGE_FAULT, pageFaultCallback);
60  // populate the physical memory map based on bootloader information
61  initPhysical(map);
62  // init our structures
63  initDirectory();
64  // identity map the first 1 MiB of RAM
65  mapEarlyMem();
66  // map in our higher-half kernel
67  mapKernel();
68  // use our new set of page tables
69  Arch::Memory::setPageDirectory(Arch::Memory::pageAlign(pageDirectoryAddress));
70  // flush the tlb and we're off to the races!
71  Arch::Memory::pagingEnable();
72 }

79 static void initPhysical(MemoryMap* map)
80 {
81  size_t freeBytes = 0;
82  size_t reservedBytes = 0;
83 
84  for (size_t i = 0; i < map->Count(); i++) {
85  auto section = map->Get(i);
86  if (section.initialized()) {
87  switch (section.type())
88  {
89  case Available:
90  physical.setFree(section);
91  freeBytes += section.size();
92  break;
93  default:
94  physical.setUsed(section);
95  reservedBytes += section.size();
96  break;
97  }
98  }
99  }
100 
101  debugf("Available memory: %zu MB\n", B_TO_MB(freeBytes));
102  debugf("Reserved memory: %zu MB\n", B_TO_MB(reservedBytes));
103  debugf("Total memory: %zu MB\n", B_TO_MB(freeBytes + reservedBytes));
104 }

107 {
108  pageDirectoryVirtual[idx] = table;

127 static void initDirectory()
128 {
129  // For every page in kernel memory
130  for (size_t i = 0; i < ARCH_PAGE_DIR_ENTRIES - 1; i++) {
131  mapKernelPageTable(i, &pageTables[i]);
132  // clear out the page tables
133  for (size_t j = 0; j < ARCH_PAGE_DIR_ENTRIES; j++) {
134  memset(&pageTables[i].pages[j], 0, sizeof(struct Arch::Memory::TableEntry));
135  }
136  }
137  // recursively map the last page table to the page directory
138  mapKernelPageTable(ARCH_PAGE_TABLE_ENTRIES - 1, (struct Arch::Memory::Table*)&pageDirectoryPhysical[0]);
139  for (size_t i = ARCH_PAGE_TABLE_ENTRIES * (ARCH_PAGE_TABLE_ENTRIES - 1); i < ARCH_PAGE_TABLE_ENTRIES * ARCH_PAGE_TABLE_ENTRIES; i++) {
140  mappedPages.Set(i);
141  }
142  // store the physical address of the page directory for quick access
143  pageDirectoryAddress = KADDR_TO_PHYS((uintptr_t)&pageDirectoryPhysical[0]);
144 }

146 void mapKernelPage(Arch::Memory::Address vaddr, Arch::Memory::Address paddr)
147 {
148  // Set the page directory entry (pde) and page table entry (pte)
149  size_t pde = vaddr.page().dirIndex;
150  size_t pte = vaddr.page().tableIndex;
151 
152  // Print a debug message to serial
153  if (is_mapping_output_enabled) {
154  debugf("map 0x%08lx to 0x%08lx, pde = 0x%08lx, pte = 0x%08lx\n", paddr.val(), vaddr.val(), pde, pte);
155  }
156 
157  // If the page's virtual address is not aligned
158  if (vaddr.page().offset) {
159  panicf("Attempted to map a non-page-aligned virtual address.\n(Address: 0x%08lX)\n", vaddr.val());
160  }
161 
162  // If the page is already mapped into memory
163  Arch::Memory::TableEntry* entry = &(pageTables[pde].pages[pte]);
164  if (entry->present) {
165  if (entry->frame == paddr.frame().index) {
166  // this page was already mapped the same way
167  return;
168  }
169 
170  panic("Attempted to map already mapped page.\n");
171  }
172  // Set the page information
173  pageTables[pde].pages[pte] = {
174  .present = 1, // The page is present
175  .readWrite = 1, // The page has r/w permissions
176  .usermode = 0, // These are kernel pages
177  .writeThrough = 0, // Disable write through
178  .cacheDisable = 0, // The page is cached
179  .accessed = 0, // The page is unaccessed
180  .dirty = 0, // The page is clean
181  .pageAttrTable = 0, // The page has no attribute table
182  .global = 0, // The page is local
183  .unused = 0, // Ignored
184  .frame = paddr.frame().index, // The last 20 bits are the frame
185  };
186  // Set the associated bit in the bitmaps
187  physical.setUsed(paddr);
188  mappedPages.Set(vaddr.frame().index);
189 }

227 void* newPage(size_t size)
228 {
229  pagingLock.lock();
230  size_t page_count = PAGE_COUNT(size);
231  size_t free_idx = findNextFreeVirtualAddress(page_count);
232  if (free_idx == SIZE_MAX)
233  return NULL;
234  for (size_t i = free_idx; i < free_idx + page_count; i++) {
235  size_t phys_page_idx = physical.findNextFreePhysicalAddress();
236  if (phys_page_idx == SIZE_MAX)
237  return NULL;
238  Arch::Memory::Address phys(phys_page_idx * ARCH_PAGE_SIZE);
239  Arch::Memory::Address vaddr(i * ARCH_PAGE_SIZE);
240  mapKernelPage(vaddr, phys);
241  }
242  pagingLock.unlock();
243  return (void*)(free_idx * ARCH_PAGE_SIZE);
244 }

246 void freePage(void* page, size_t size)
247 {
248  pagingLock.lock();
249  size_t page_count = PAGE_COUNT(size);
250  Arch::Memory::Address addr((uintptr_t)page);
251  for (size_t i = addr.page().tableIndex; i < addr.page().tableIndex + page_count; i++) {
252  mappedPages.Reset(i);
253  // this is the same as the line above
254  struct Arch::Memory::TableEntry* pte = &(pageTables[i / ARCH_PAGE_TABLE_ENTRIES].pages[i % ARCH_PAGE_TABLE_ENTRIES]);
255  // the frame field is actually the page frame's index basically it's frame 0, 1...(2^21-1)
256  physical.setFree(pte->frame);
257  // zero it out to unmap it
258  memset(pte, 0, sizeof(struct Arch::Memory::TableEntry));
259  // clear that tlb
260  Arch::Memory::pageInvalidate(page);
261  }
262  pagingLock.unlock();
263 }
