61 void init(Memory::MemoryMap* map)
62 {
63  for (size_t i = 0; i < map->Count(); i++) {
64  auto section = map->Get(i);
65  if (section.Initialized()) {
66  debugf("[%s]\t0x%08X - 0x%08X\n", section.TypeString(), section.Base(), section.Size());
67  }
68  }
69  // we can set breakpoints or make a futile attempt to recover.
70  register_interrupt_handler(ISR_PAGE_FAULT, pageFaultCallback);
71  // init our structures
72  initDirectory();
73  // identity map the first 1 MiB of RAM
74  mapEarlyMem();
75  // map in our higher-half kernel
76  mapKernel();
77  // use our new set of page tables
78  setPageDirectory(Arch::Memory::pageAlign(pageDirectoryAddress));
79  // flush the tlb and we're off to the races!
80  Arch::Memory::pagingEnable();
81 }

88 static inline void mapKernelPageTable(uint32_t idx, struct Table* table)
89 {
90  pageDirectoryVirtual[idx] = table;
91  pageDirectoryPhysical[idx] = {
92  .present = 1,
93  .readWrite = 1,
94  .usermode = 0,
95  .writeThrough = 0,
96  .cacheDisable = 0,
97  .accessed = 0,
98  .ignoredA = 0,
99  .size = 0,
100  .ignoredB = 0,
101  // compute the physical address of this page table the virtual address is obtained with the & operator and
102  // the offset is applied from the load address of the kernel we must shift it over 12 bits because we only
103  // care about the highest 20 bits for the page table
104  .tableAddr = KADDR_TO_PHYS((uint32_t)table) >> 12
105  };
106 }

108 static void initDirectory()
109 {
110  // For every page in kernel memory
111  for (int i = 0; i < PAGE_ENTRIES - 1; i++) {
112  mapKernelPageTable(i, &pageTables[i]);
113  // clear out the page tables
114  for (int j = 0; j < PAGE_ENTRIES; j++) {
115  memset(&pageTables[i].pages[j], 0, sizeof(struct TableEntry));
116  }
117  }
118  // recursively map the last page table to the page directory
119  mapKernelPageTable(PAGE_ENTRIES - 1, (struct Table*)&pageDirectoryPhysical[0]);
120  for (uint32_t i = PAGE_ENTRIES * (PAGE_ENTRIES - 1); i < PAGE_ENTRIES * PAGE_ENTRIES; i++) {
121  mappedPages.Set(i);
122  }
123  // store the physical address of the page directory for quick access
124  pageDirectoryAddress = KADDR_TO_PHYS((uint32_t)&pageDirectoryPhysical[0]);
125 }

127 void mapKernelPage(union Address vaddr, union Address paddr)
128 {
129  // Set the page directory entry (pde) and page table entry (pte)
130  uint32_t pde = vaddr.page.dirIndex;
131  uint32_t pte = vaddr.page.tableIndex;
132  // If the page's virtual address is not aligned
133  if (vaddr.page.offset != 0) {
134  PANIC("Attempted to map a non-page-aligned virtual address.\n");
135  }
136  TableEntry* entry = &(pageTables[pde].pages[pte]);
137  // Print a debug message to serial
138  if (is_mapping_output_enabled) {
139  debugf("map 0x%08x to 0x%08x, pde = 0x%08x, pte = 0x%08x\n", paddr.val, vaddr.val, pde, pte);
140  }
141  // If the page is already mapped into memory
142  if (entry->present) {
143  if (entry->frame == paddr.frame.index) {
144  // this page was already mapped the same way
145  return;
146  }
147 
148  PANIC("Attempted to map already mapped page.\n");
149  }
150  // Set the page information
151  pageTables[pde].pages[pte] = {
152  .present = 1, // The page is present
153  .readWrite = 1, // The page has r/w permissions
154  .usermode = 0, // These are kernel pages
155  .writeThrough = 0, // Disable write through
156  .cacheDisable = 0, // The page is cached
157  .accessed = 0, // The page is unaccessed
158  .dirty = 0, // The page is clean
159  .pageAttrTable = 0, // The page has no attribute table
160  .global = 0, // The page is local
161  .unused = 0, // Ignored
162  .frame = paddr.frame.index, // The last 20 bits are the frame
163  };
164  // Set the associated bit in the bitmaps
165  mappedMemory.Set(paddr.frame.index);
166  mappedPages.Set(vaddr.frame.index);
167 }

219 void* newPage(uint32_t size)
220 {
221  pagingLock.Lock();
222  uint32_t page_count = (size / ARCH_PAGE_SIZE) + 1;
223  uint32_t free_idx = findNextFreeVirtualAddress(page_count);
224  if (free_idx == SIZE_MAX)
225  return NULL;
226  for (uint32_t i = free_idx; i < free_idx + page_count; i++) {
227  uint32_t phys_page_idx = findNextFreePhysicalAddress();
228  if (phys_page_idx == SIZE_MAX)
229  return NULL;
230  union Address phys = {
231  .val = phys_page_idx * ARCH_PAGE_SIZE,
232  };
233  mapKernelPage(ADDR((uint32_t)i * ARCH_PAGE_SIZE), phys);
234  }
235  pagingLock.Unlock();
236  return (void*)(free_idx * ARCH_PAGE_SIZE);
237 }

239 void freePage(void* page, uint32_t size)
240 {
241  pagingLock.Lock();
242  uint32_t page_count = (size / ARCH_PAGE_SIZE) + 1;
243  uint32_t page_index = (uint32_t)page >> 12;
244  for (uint32_t i = page_index; i < page_index + page_count; i++) {
245  mappedPages.Reset(i);
246  // this is the same as the line above
247  struct TableEntry* pte = &(pageTables[i / PAGE_ENTRIES].pages[i % PAGE_ENTRIES]);
248  // the frame field is actually the page frame's index basically it's frame 0, 1...(2^21-1)
249  mappedMemory.Reset(pte->frame);
250  // zero it out to unmap it
251  *pte = { /* Zero */ };
252  // clear that tlb
253  Arch::Memory::pageInvalidate(page);
254  }
255  pagingLock.Unlock();
256 }
