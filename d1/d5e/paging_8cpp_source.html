83 static inline void map_kernel_page_table(uint32_t pd_idx, struct page_table* table)
84 {
85  page_dir_virt[pd_idx] = table;
86  page_dir_phys[pd_idx] = {
87  .present = 1,
88  .readWrite = 1,
89  .usermode = 0,
90  .writeThrough = 0,
91  .cacheDisable = 0,
92  .accessed = 0,
93  .ignoredA = 0,
94  .size = 0,
95  .ignoredB = 0,
96  // compute the physical address of this page table
97  // the virtual address is obtained with the & operator and
98  // the offset is applied from the load address of the kernel
99  // we must shift it over 12 bits because we only care about
100  // the highest 20 bits for the page table
101  .tableAddr = KADDR_TO_PHYS((uint32_t)table) >> 12
102  };
103 }

105 static void paging_init_dir()
106 {
107  // For every page in kernel memory
108  for (int i = 0; i < PAGE_ENTRIES - 1; i++) {
109  map_kernel_page_table(i, &page_tables[i]);
110  // clear out the page tables
111  for (int j = 0; j < PAGE_ENTRIES; j++) {
112  page_tables[i].pages[j] = (struct page_table_entry) { /* ZERO */ };
113  }
114  }
115  // recursively map the last page table to the page directory
116  map_kernel_page_table(PAGE_ENTRIES - 1, (struct page_table*)&page_dir_phys[0]);
117  for (uint32_t i = PAGE_ENTRIES * (PAGE_ENTRIES - 1); i < PAGE_ENTRIES * PAGE_ENTRIES; i++) {
118  mapped_pages.Set(i);
119  }
120  // store the physical address of the page directory for quick access
121  page_dir_addr = KADDR_TO_PHYS((uint32_t)&page_dir_phys[0]);
122 }

124 void map_kernel_page(union address vaddr, union address paddr)
125 {
126  // Set the page directory entry (pde) and page table entry (pte)
127  uint32_t pde = vaddr.page.dirIndex;
128  uint32_t pte = vaddr.page.tableIndex;
129  // If the page's virtual address is not aligned
130  if (vaddr.page.offset != 0) {
131  PANIC("Attempted to map a non-page-aligned virtual address.\n");
132  }
133  page_table_entry* entry = &(page_tables[pde].pages[pte]);
134  // Print a debug message to serial
135  if (is_mapping_output_enabled) {
136  debugf("map 0x%08x to 0x%08x, pde = 0x%08x, pte = 0x%08x\n", paddr.val, vaddr.val, pde, pte);
137  }
138  // If the page is already mapped into memory
139  if (entry->present) {
140  if (entry->frame == paddr.frame.index) {
141  // this page was already mapped the same way
142  return;
143  }
144 
145  PANIC("Attempted to map already mapped page.\n");
146  }
147  // Set the page information
148  page_tables[pde].pages[pte] = {
149  .present = 1, // The page is present
150  .readWrite = 1, // The page has r/w permissions
151  .usermode = 0, // These are kernel pages
152  .writeThrough = 0, // Disable write through
153  .cacheDisable = 0, // The page is cached
154  .accessed = 0, // The page is unaccessed
155  .dirty = 0, // The page is clean
156  .pageAttrTable = 0, // The page has no attribute table
157  .global = 0, // The page is local
158  .unused = 0, // Ignored
159  .frame = paddr.frame.index, // The last 20 bits are the frame
160  };
161  // Set the associated bit in the bitmaps
162  mapped_mem.Set(paddr.frame.index);
163  mapped_pages.Set(vaddr.frame.index);
164 }

217  * map in a new page. if you request less than one page, you will get exactly one page

219 void* get_new_page(uint32_t size)
220 {
221  mutex_paging.Lock();
222  uint32_t page_count = (size / PAGE_SIZE) + 1;
223  uint32_t free_idx = find_next_free_virt_addr(page_count);
224  if (free_idx == SIZE_MAX)
225  return NULL;
226  for (uint32_t i = free_idx; i < free_idx + page_count; i++) {
227  uint32_t phys_page_idx = find_next_free_phys_page();
228  if (phys_page_idx == SIZE_MAX)
229  return NULL;
230  union address phys = {
231  .val = phys_page_idx * PAGE_SIZE,
232  };
233  map_kernel_page(ADDR((uint32_t)i * PAGE_SIZE), phys);
234  }
235  mutex_paging.Unlock();
236  return (void*)(free_idx * PAGE_SIZE);
237 }

239 void free_page(void* page, uint32_t size)
240 {
241  mutex_paging.Lock();
242  uint32_t page_count = (size / PAGE_SIZE) + 1;
243  uint32_t page_index = (uint32_t)page >> 12;
244  for (uint32_t i = page_index; i < page_index + page_count; i++) {
245  mapped_pages.Reset(i);
246  // this is the same as the line above
247  struct page_table_entry* pte = &(page_tables[i / PAGE_ENTRIES].pages[i % PAGE_ENTRIES]);
248  // the frame field is actually the page frame's index
249  // basically it's frame 0, 1...(2^21-1)
250  mapped_mem.Reset(pte->frame);
251  // zero it out to unmap it
252  *pte = { /* Zero */ };
253  // clear that tlb
254  Arch::pageInvalidate(page);
255  }
256  mutex_paging.Unlock();
257 }
