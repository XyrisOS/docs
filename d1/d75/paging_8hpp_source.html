55 /**
56  * @brief Page table entry defined in accordance to the
57  * Intel Developer Manual Vol. 3a p. 4-12
58  *
59  */
60 struct page_table_entry
61 {
62  uint32_t present : 1; // Page present in memory
63  uint32_t readWrite : 1; // Read-only if clear, readwrite if set
64  uint32_t usermode : 1; // Supervisor level only if clear
65  uint32_t writeThrough : 1; // Page level write through
66  uint32_t cacheDisable : 1; // Disables TLB caching of page entry
67  uint32_t accessed : 1; // Has the page been accessed since last refresh?
68  uint32_t dirty : 1; // Has the page been written to since last refresh?
69  uint32_t pageAttrTable : 1; // Page attribute table (memory cache control)
70  uint32_t global : 1; // Prevents the TLB from updating the address
71  uint32_t unused : 3; // Amalgamation of unused and reserved bits
72  uint32_t frame : 20; // Frame address (shifted right 12 bits)
73 };
