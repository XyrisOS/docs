54 /**
55  * @brief Page table entry defined in accordance to the
56  * Intel Developer Manual Vol. 3a p. 4-12
57  *
58  */
59 struct page_table_entry
60 {
61  uint32_t present : 1; // Page present in memory
62  uint32_t read_write : 1; // Read-only if clear, readwrite if set
63  uint32_t usermode : 1; // Supervisor level only if clear
64  uint32_t write_through : 1; // Page level write through
65  uint32_t cache_disable : 1; // Disables TLB caching of page entry
66  uint32_t accessed : 1; // Has the page been accessed since last refresh?
67  uint32_t dirty : 1; // Has the page been written to since last refresh?
68  uint32_t page_att_table : 1; // Page attribute table (memory cache control)
69  uint32_t global : 1; // Prevents the TLB from updating the address
70  uint32_t unused : 3; // Amalgamation of unused and reserved bits
71  uint32_t frame : 20; // Frame address (shifted right 12 bits)
72 };
