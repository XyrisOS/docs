27 void find_primes(void)
28 {
29  for (prime_current = 2; prime_current < PRIME_MAX_SQRT; prime_current++) {
30  if (!map.Test(prime_current)) continue;
31  for (size_t j = prime_current * prime_current; j < PRIME_MAX; j += prime_current) {
