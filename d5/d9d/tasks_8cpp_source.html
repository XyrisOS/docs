96 static void _discover_cpu_speed()
97 {
98  uint32_t curr_tick = timer_tick;
99  uint64_t curr_rtsc = __rdtsc();
100  while (timer_tick != curr_tick + 1) { }
101  curr_rtsc = __rdtsc() - curr_rtsc;
102  _instr_per_ns = curr_rtsc / 1000000;
103  // will be inaccurate, but it's the best we can do in these circumstances
104  if (_instr_per_ns == 0) _instr_per_ns = 1;

217 static void _enqueue_task(struct tasklist *list, task *task)
218 {
219  if (list->head == NULL) {
220  list->head = task;
221  }
222  if (list->tail != NULL) {
223  // the current last task's next pointer will be this task
224  list->tail->next = task;
225  }
226  // and now this task becomes the last task
227  task->next = NULL;
228  list->tail = task;
229 }

231 static struct task *_dequeue_task(struct tasklist *list)
232 {
233  struct task *task;
234  if (list->head == NULL) {
235  // can't dequeue if there's not anything there
236  return NULL;
237  }
238  // the head of the list is the next item
239  task = list->head;
240  // the new head is the next task
241  list->head = task->next;
242  // so null its previous pointer
243  if (list->head == NULL) {
244  // if there are no more items in the list, then
245  // the last item in the list will also be null
246  list->tail = NULL;
247  }
248  // it doesn't make sense to have a next when it's not in a list
249  task->next = NULL;
250  return task;
251 }

253 static void _remove_task(struct tasklist *list, struct task *task, struct task *previous)
254 {
255  // if this is true, something's not right...
256  if (previous != NULL && previous->next != task) {
257  PANIC("Bogus arguments to _remove_task.\n");
258  }
259  // update the head if necessary
260  if (list->head == task) {
261  list->head = task->next;
262  }
263  // update the tail if necessary
264  if (list->tail == task) {
265  list->tail = previous;
266  }
267  // update the previous task if necessary
268  if (previous != NULL) {
269  previous->next = task->next;
270  }
271  // it's not in any list anymore, so clear its next pointer
272  task->next = NULL;
273 }

285 struct task *tasks_new(void (*entry)(void), struct task *storage, task_state state, const char *name)
286 {
287  struct task *new_task = storage;
288  if (storage == NULL) {
289  // allocate memory for our task structure
290  new_task = (struct task*)malloc(sizeof(struct task));
291  // panic if the alloc fails (we have no fallback)
292  if (new_task == NULL) {
293  PANIC("Unable to allocate memory for new task struct.\n");
294  }
295  }
296  // allocate a page for this stack (we might change this later)
297  uint8_t *stack = (uint8_t *)get_new_page(PAGE_SIZE - 1);
298  if (stack == NULL) PANIC("Unable to allocate memory for new task stack.\n");
299  // remember, the stack grows up
300  void *stack_pointer = stack + PAGE_SIZE;
301  // a null stack frame to make the panic screen happy
302  _stack_push_word(&stack_pointer, 0);
303  // the last thing to happen is the task stopping function
304  _stack_push_word(&stack_pointer, (size_t)_task_stopping);
305  // next entry is the main function to call (the start of the task)
306  _stack_push_word(&stack_pointer, (size_t)entry);
307  // when this task is started, the CPU will pop off this value which will become the new EIP
308  // we push this function to allow some setup code to be run from within the context of the new task
309  _stack_push_word(&stack_pointer, (size_t)_task_starting);
310  // our task switching code is going to pop four values off of the stack before returning
311  _stack_push_word(&stack_pointer, 0);
312  _stack_push_word(&stack_pointer, 0);
313  _stack_push_word(&stack_pointer, 0);
314  _stack_push_word(&stack_pointer, 0);
315  new_task->stack_top = (uintptr_t)stack_pointer;
316  new_task->page_dir = get_phys_page_dir();
317  new_task->next = NULL;
318  new_task->state = state;
319  new_task->time_used = 0;
320  new_task->name = name;
321  new_task->alloc = storage == NULL ? ALLOC_DYNAMIC : ALLOC_STATIC;
322  if (state == TASK_READY) {
323  _tasks_enqueue_ready(new_task);
324  }
325  TASK_ACTION("create task", new_task);
326  return new_task;
327 }

341 static void _schedule()
342 {
343  if (_scheduler_postpone_count != 0) {
344  // don't schedule if there's more work to be done
345  _scheduler_postponed = true;
346  return;
347  }
348  if (current_task == NULL) {
349  // we are currently idling and will schedule at a later time
350  return;
351  }
352  // get the next task
353  struct task *task = _tasks_dequeue_ready();
354  // don't need to do anything if there's nothing ready to run
355  if (task == NULL) {
356  if (current_task->state == TASK_RUNNING) {
357  // still running the same task
358  // but also reset the time slice counter
359  _time_slice_remaining = TIME_SLICE_SIZE;
360  return;
361  }
362  // disable time slices because there are no tasks available to run
363  _time_slice_remaining = 0;
364  // count the time that this task ran for
365  tasks_update_time();
366  /*** idle ***/
367  // borrow this task to return to once we're not idle anymore
368  struct task *borrowed = current_task;
369  // set the current task to null to indicate an idle state
370  current_task = NULL;
371  _idle_start = _get_cpu_time_ns();
372  do {
373  // enable interrupts to process timer and other events
374  asm ("sti");
375  // immediately halt the CPU
376  asm ("hlt");
377  // disable interrupts to restore our lock
378  asm ("cli");
379  // check if there's a task ready to be run
380  } while (task = _tasks_dequeue_ready(), task == NULL);
381  // count the time we spent idling
382  tasks_update_time();
383  // reset the current task
384  current_task = borrowed;
385  _idle_start = _idle_start - _get_cpu_time_ns();
386  _idle_time += _idle_start;
387  } else {
388  // just do time accounting once
389  tasks_update_time();
390  }
391  // reset the time slice because a new task is being scheduled
392  _time_slice_remaining = TIME_SLICE_SIZE;
393 #ifdef DEBUG
394  RS232::printf("switching to ");
395  _print_task(task);
396 #endif
397  // reset the last "timer time" since the time slice was reset
398  _last_timer_time = _get_cpu_time_ns();
399  // switch to the task
400  tasks_switch_to(task);
401 }

445 static void _on_timer()
446 {
447  _aquire_scheduler_lock();
448 
449  struct task *pre = NULL;
450  struct task *task = tasks_sleeping.head;
451  struct task *next;
452  bool need_schedule = false;
453  uint64_t time = _get_cpu_time_ns();
454  uint64_t time_delta;
455 
456  while (task != NULL) {
457  next = task->next;
458  if (time >= task->wakeup_time) {
459  //RS232::printf("timer: waking sleeping task\n");
460  _remove_task(&tasks_sleeping, task, pre);
461  _wakeup(task);
462  task->next = NULL;
463  need_schedule = true;
464  } else {
465  pre = task;
466  }
467  task = next;
468  }
469 
470  if (_time_slice_remaining != 0) {
471  time_delta = time - _last_timer_time;
472  _last_timer_time = time;
473  if (time_delta >= _time_slice_remaining) {
474  // schedule (and maybe pre-empt)
475  // the schedule function will reset the time slice
476  //RS232::printf("timer: time slice expired\n");
477  need_schedule = true;
478  } else {
479  // decrement the time slice counter
480  _time_slice_remaining -= time_delta;
481  }
482  }
483 
484  if (need_schedule) {
485  _schedule();
486  }
487 
488  _release_scheduler_lock();
489 }
