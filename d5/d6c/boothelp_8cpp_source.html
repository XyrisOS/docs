35 stivale2_mmap_helper(void* baseaddr)
36 {
37  struct stivale2_tag* tag = (struct stivale2_tag*)baseaddr;
38 
39  while (tag) {
40  switch (tag->identifier) {
41  case STIVALE2_STRUCT_TAG_MEMMAP_ID: {
42  auto memmap = (struct stivale2_struct_tag_memmap*)tag;
43  for (size_t i = 0; i < memmap->entries; i++) {
44  switch (memmap->memmap[i].type) {
45  case STIVALE2_MMAP_BOOTLOADER_RECLAIMABLE:
46  if (((uintptr_t)memmap->memmap[i].base) == (uintptr_t)baseaddr) {
47  return (uintptr_t)memmap->memmap[i].length;
48  }
49  break;
50 
51  default:
52  break;
53  }
54  }
55  }
56  default:
57  break;
58  }
59  tag = (struct stivale2_tag*)tag->next;
60  }
61  // If we get here, there's a problem so we will just
62  // return 0 and panic in boot.S
63  return 0;
